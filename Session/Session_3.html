6. Console Output: Variables, Operators, Array

Shows variable assignments, skills array loop, and several arithmetic operations on age. 

        let name = "Rishu";
        let age = 20;
        const isStudent = true;
        let skills = ["Coding", "Singing", "Gaming"];

        console.log("Name: " + name);
        console.log("Age: " + age);
        console.log("Is Student: " + isStudent);
        skills.forEach(skill => console.log("Single skill: " + skill));
        console.log("Using operators on age:");
        console.log("age + 5:", age +5); 
        console.log("age - 5:", age - 5);
        console.log("age * 2:", age * 2);
        console.log("age / 2:", age / 2);
